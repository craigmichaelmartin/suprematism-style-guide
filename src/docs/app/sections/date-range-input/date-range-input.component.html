Date Range Input

Installation
  
  // install the repo
  npm install CINBCUniversal/suprematism-date-range-input --save

  // add it to a module (app.module.ts)
  import {{'{'}} DateRangeInputModule } from 'suprematism-date-range-input';

  @NgModule({{'{'}}
    ...
    imports: [
      ...
      DateRangeInputModule
    ],
    ...
  })
Pick a Range
  
    {{'<'}}supre-date-range-input
      [value]="value"
      [min]="minDate"
      [max]="maxDate"
      [minDisplayWeeks]="minDisplayWeeks"
      [startDayOfWeek]="startDayOfWeek"
      (valueChange)="value = $event"
      >
    {{'<'}}/supre-date-range-input>
