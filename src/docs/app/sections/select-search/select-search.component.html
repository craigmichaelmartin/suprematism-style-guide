Select Search

Installation
  
  // install the repo
  npm install CINBCUniversal/suprematism-select-search --save

  // install dependencies
  npm install ag-grid ag-grid-angular ag-grid-enterprise --save

  // add it to a module (app.module.ts)
  import {{'{'}} SelectSearchModule } from 'suprematism-select-search';

  @NgModule({{'{'}}
    ...
    imports: [
      ...
      SelectSearchModule
    ],
    ...
  })
This is a select search.
  
    {{'<'}}supre-select-search [data]="data1"
                         unselectedText="SELECT AN AUDIENCE"
                         [filterFunc]="stringFilter"
                         [displayFormat]="displayFormat1">
    {{'<'}}/supre-select-search>
This select seach will still close if the input is focused.
  
    <{{'<'}}supre-select-search [data]="data1"
                         unselectedText="SELECT AN AUDIENCE"

    {{'<'}}/supre-select-search>
This select seach filters on id.
  
    {{'<'}}supre-select-search [data]="data1"
                         unselectedText="SELECT AN AUDIENCE"

    {{'<'}}/supre-select-search>
