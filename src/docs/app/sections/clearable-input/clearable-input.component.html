Clearable Input

Installation

  npm install CINBCUniversal/suprematism-clearable-input --save

  // add it to a module (app.module.ts)
  import {{'{'}} ClearableInputModule } from 'suprematism-clearable-input';

  @NgModule({{'{'}}
    ...
    imports: [
      ...
      ClearableInputModule
    ],
    ...
  })

      placeholder="Search"

    {{'<'}}/supre-clearable-input>
