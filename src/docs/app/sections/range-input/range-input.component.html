Range Input

Installation
  
  // install the repo
  npm install CINBCUniversal/suprematism-range-input --save

  // add it to a module (app.module.ts)
  import {{'{'}} RangeInputModule } from 'suprematism-range-input';

  @NgModule({{'{'}}
    ...
    imports: [
      ...
      RangeInputModule
    ],
    ...
  })
With Units Toggle
range ({{range1.units}}) : {{range1.min || '--'}} to {{range1.max || '--'}} 
  
    {{'<'}}supre-range-input
      minPlaceholder="Enter Min"
      maxPlaceholder="Enter Max"
      (valueChange)="range1 = $event"
      >
    {{'<'}}/supre-range-input>
Without Units Toggle
range : {{range2.min || '--'}} to {{range2.max || '--'}} 
  
    {{'<'}}supre-range-input
      [showUnits]="false"
      minPlaceholder="Enter Min"
      maxPlaceholder="Enter Max"
      (valueChange)="range2 = $event"
      >
    {{'<'}}/supre-range-input>
