  @import "../node_modules/suprematism-style-guide/dist/utils/_colors.scss";
  

  
  //////////////////////////////////////////////////////////////////
  // SASS/SCSS List of all the colors
  //////////////////////////////////////////////////////////////////

  $supre-colors: (

    ('ice', $ice),
    ('off-white', $off-white),
    ('gray', $gray),
    ('lemon', $lemon),
    ('blue-gray', $blue-gray),
    ('aqua', $aqua),
    ('purple', $purple),
    ('mid-gray', $mid-gray),
    ('blue', $blue),
    ('green', $green),
    ('lite-green', $lite-green),
    ('white', $white),
    ('black', $black),
    ('lite-gray', $lite-gray),
    ('active-stroke', $active-stroke),
    ('disabled-stroke', $disabled-stroke),
    ('soft-green', $soft-green),
    ('shark-skin', $shark-skin),
    ('red', $red),

    ('sequence-1', $sequence-1),
    ('sequence-2', $sequence-2),
    ('sequence-3', $sequence-3),
    ('sequence-4', $sequence-4),

    ('viz-aqua', $viz-aqua),
    ('viz-purple', $viz-purple),
    ('viz-lime', $viz-lime),
    ('viz-ocean', $viz-ocean),
    ('viz-orange', $viz-orange),
    ('viz-blue', $viz-blue),
    ('viz-yellow', $viz-yellow)

  );


  //////////////////////////////////////////////////////////////////
  // Mixins
  // These mixins can help with avoiding specificity issues
  //////////////////////////////////////////////////////////////////

  // append all background color classes
  @mixin appendBackgroundColorClasses();
  // add only the supplied background color classes
  @mixin appendBackgroundColorClasses('aqua', 'lemon', 'viz-purple');


  // append all text color classes
  @mixin appendTextColorClasses();
  // add only the supplied text color classes
  @mixin appendTextColorClasses('blue', 'sequence-2', 'lite-gray', 'mid-gray');


  // append all border color classes
  @mixin appendBorderColorClasses();
  // add only the supplied border color classes
  @mixin appendBorderColorClasses('shark-skin', 'off-white');
