Margin SCSS Place Holders

This will add a margin style to any class you use it on.
Spacing placehoders are a Utility, add it to your SCSS file from here:

  
  @import "../node_modules/suprematism-style-guide/dist/utils/_spacing.scss";
  

  
  // use the @extend directive, placholders start with '%', not a '.'

  @extend %_mxs;  // margin: ...px;
  @extend %_mtxs; // margin-top: ...px;
  @extend %_mrxs; // margin-right: ...px;
  @extend %_mbxs; // margin-bottom: ...px;
  @extend %_mlxs; // margin-left: ...px;

  @extend %_ms;  // margin: ...px;
  @extend %_mts; // margin-top: ...px;
  @extend %_mrs; // margin-right: ...px;
  @extend %_mbs; // margin-bottom: ...px;
  @extend %_mls; // margin-left: ...px;

  @extend %_mm;  // margin: ...px;
  @extend %_mtm; // margin-top: ...px;
  @extend %_mrm; // margin-right: ...px;
  @extend %_mbm; // margin-bottom: ...px;
  @extend %_mlm; // margin-left: ...px;

  @extend %_ml;  // margin: ...px;
  @extend %_mtl; // margin-top: ...px;
