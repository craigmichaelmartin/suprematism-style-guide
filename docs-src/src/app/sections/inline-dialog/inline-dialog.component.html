Inline Dialog

Installation
  
  // install the repo
  npm install CINBCUniversal/suprematism-popover --save
  
  // add it to a module (app.module.ts)
  import {{'{'}} PopoverModule } from 'suprematism-popover';

  @NgModule({{'{'}}
    ...
    imports: [
      ...
      PopoverModule
    ],
    ...
  })
Simple Example

Here is the text that could go inside the inline dialog?

 
  
  {{'<'}}!-- Bind the 'supre-inline-dialog' instance to anything, a button in this case -->
  {{'<'}}button class="-action" [supreInlineDialog]="dialogInstance">Show Dialog{{'<'}}/button>

  {{'<'}}!-- Add a 'supre-inline-dialog' instance -->
  {{'<'}}supre-inline-dialog #dialogInstance title="Title of the Dialog">
    {{'<'}}p class="_mbm">Here is the text that could go inside the inline dialog?{{'<'}}/p>
    {{'<'}}div class="flex justify-end">
      {{'<'}}!-- We can directly access the 'supre-inline-dialog' api -->
      {{'<'}}button class="-cancel-md" (click)="dialogInstance.hide()">Cancel{{'<'}}/button>
      {{'<'}}button class="-dialog">Submit{{'<'}}/button>
    {{'<'}}/div>
  {{'<'}}/supre-inline-dialog>
Combined with Tooltip
 

Are you sure you want to clear all attributes and conditions?

 
  
  {{'<'}}!-- Condition Bar -->
  {{'<'}}div class="flex justify-start">
    {{'<'}}div class="flex _b1 border-gray">
      {{'<'}}div class="flex-auto bg-ice border-white _br1"> {{'<'}}/div>
      {{'<'}}button class="-toolbar" disabled="disabled">{{'<'}}span class="u-supre-icon u-supre-icon--add-create">{{'<'}}/span> Condition{{'<'}}/button>
      {{'<'}}button class="-toolbar" supreTooltip="Clear All Conditions" [supreInlineDialog]="clearAllConditions" [onHover]="false">{{'<'}}span class="u-supre-icon u-supre-icon--refresh-clear">{{'<'}}/span> Clear All{{'<'}}/button>
    {{'<'}}/div>
  {{'<'}}/div>

  {{'<'}}!-- Clear Dialog Box -->
  {{'<'}}supre-inline-dialog #clearAllConditions title="Clear All Conditions" placement="auto bottom">
    {{'<'}}p class="_mbm">Are you sure you want to clear all attributes and conditions?{{'<'}}/p>
    {{'<'}}div class="flex justify-end">
      {{'<'}}button class="-cancel-md" (click)="clearAllConditions.hide()">Cancel{{'<'}}/button>
      {{'<'}}button class="-dialog bg-aqua">{{'<'}}span class="u-supre-icon u-supre-icon--refresh-clear">{{'<'}}/span> Clear All{{'<'}}/button>
    {{'<'}}/div>
  {{'<'}}/supre-inline-dialog>
