API
  
  /** 
   * Classes and Interfaces 
   */
  import {{'{'}}
    ModalEvent, // an interface for the event emitted from the modal
    ModalComponent // the component class
  } from 'suprematism-modal';


  /**
   * Event (emitter)
   */
  {{'<'}}supre-modal (onChange)="yourMethod($event: ModalEvent);">...{{'<'}}/supre-modal>


  /**
   * ModalEvent Interface
   */
  export interface ModalEvent {{'{'}}
    // tells you what method/action was called
    methodCalled: 'open' | 'close' | null;
    // the current state
    isOpen: boolean;
    // the value of the 'id' property/attribute on the {{'<'}}supre-modal> element, null if there is none
    id: string | null;
  }


  /** 
   * ModalComponent Class
   */
  ModalComponent.open();  // emits: {{'{'}} methodCalled: 'open', isOpen: true, id: null }
  ModalComponent.close(); // emits: {{'{'}} methodCalled: 'close', isOpen: false, id: null }
  // NOTE: if {{'<'}}supre-modal id="myModal">...{{'<'}}/supre-modal> then the id would be: 'myModal'
